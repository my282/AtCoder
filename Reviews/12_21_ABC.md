### 問題概要

### 提出コード
```C
#include <stdio.h>

int maxValue(int array[], int n) {
  int max;
  max = array[0];
  for (int i = 0; i < n; i++) {
    if (array[i] > max) {
      max = array[i];
    }
  }
  return max;
}

int main(void) {
  int h, w, n;
  int count_buffer[3] = {0};
  int answer;
  scanf("%d", &h);
  scanf("%d", &w);
  scanf("%d", &n);

  int A[3][5];
  int B[90];
  for (int j = 0; j < h; j++) {
    for (int i = 0; i < w; i++) {
      scanf("%d", &A[j][i]);
    }
  }

  for (int i = 0; i < n; i++) {
    scanf("%d", &B[i]);
  }

  for (int j = 0; j < h; j++) {
    for (int i = 0; i < w; i++) {
      for (int k = 0; k < n; k++) {
        if (A[j][i] == B[k]) {
          count_buffer[j]++;
        }
      }
    }
  }
  answer = maxValue(count_buffer, h);
  printf("%d", answer);
  return 0;
}

```
